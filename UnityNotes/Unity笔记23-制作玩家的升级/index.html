Unity笔记23-制作玩家的升级

Created2021-07-30|Updated2021-07-31|Unity学习笔记
|Post View:|Comments:

在本章中我们来制作玩家的经验系统，并附有升级功能

扩展CharacterData_SO

我们首先打开CharacterData_SO.cs添加一些额外的属性进去，同时写一个升级时处理的方法

[CreateAssetMenu(fileName = "New Data",menuName = "Character States/Data")]
public class CharacterData_SO : ScriptableObject{
    [Header("States Info（属性信息）")]
    public int maxHealth;//最大生命值
    public int currentHealth;//当前生命值
    public int baseDefence;//基础防御值
    public int currentDefence;//当前防御值


    [Header("Kill（击杀信息）")]
    public int killPoint;//击杀后掉落的经验值


    [Header("Level（等级信息）")]
    public int currentLevel;//当前等级
    public int maxLevel;//最大等级数
    public int baseExp;//基础经验值
    public int currentExp;//当前经验值
    public float levelBuff;//每次升级整体属性提升的百分比

    public float LevelMultiplier{
        //升级提升属性加的倍率
        get{return 1 + (currentLevel-1) * levelBuff;}
    }

    public void UpdateExp(int point){
        currentExp += point;//把外界的经验加进来
        if(currentExp >= baseExp)
            LevelUp();
    }

    private void LevelUp(){
        //这个方法中添加升级时想提升的数据

        //将current+1限制在[0,maxLevel]闭区间中
        currentLevel = Mathf.Clamp(currentLevel+1,0,maxLevel);

        //将经验和生命值按照每一等级不同倍率增长（也可以使用固定值level buff）
        baseExp += (int)(baseExp*LevelMultiplier);
        maxHealth = (int)(maxHealth * LevelMultiplier);
        currentHealth = maxHealth;

    }
}

接着再去到CharacterStates.cs做最后的修改

public void TakeDamage(CharacterStates attacker,CharacterStates defener){
    //计算伤害数值
    //当前攻击力减去目标的防御力，如果防御力过高就是0伤害
    int damage = Mathf.Max(attacker.CurrentDamage()-defener.currentDefence,0);
    currentHealth = Mathf.Max(currentHealth-damage,0);//血量也是最小值锁在0
    Debug.LogFormat("{0}攻击了{1}，造成了{2}点伤害",attacker.gameObject.name,defener.gameObject.name,damage);

    if(attacker.isCritical)//判断被打者是否受到暴击并且播放受伤动画
        defener.GetComponent<Animator>().SetTrigger("Hit");

    //将来要做的事情
    //血量条UI
    UpdateHealthBarOnAttack?.Invoke(currentHealth,MaxHealth);//如果事件不为空则调用
    //打死怪后加经验
    if(defener.currentHealth <= 0)
        attacker.characterData.UpdateExp(defener.characterData.killPoint);
}


public void TakeDamage(int damage,CharacterStates defener){
    //石头反击石头人的攻击方式
    int dam = Mathf.Max(damage - defener.currentDefence,0);
    defener.currentHealth = Mathf.Max(defener.currentHealth - dam,0);

    //血条事件
    UpdateHealthBarOnAttack?.Invoke(currentHealth,MaxHealth);//如果事件不为空则调用

    //TODO:加经验
    //可以通过GameManager来访问玩家的属性
    if (defener.currentHealth <= 0)
    GameManager.Instance.playerStates.characterData.UpdateExp(defener.characterData.killPoint);
}
