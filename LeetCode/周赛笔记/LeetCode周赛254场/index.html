class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        int nLen = nums.size();
        sort(nums.begin(),nums.end());
        
        int ans = 0;
        vector<int> res(nLen);
        for(int i=0;i<nLen;i+=2)
            res[i]=nums[ans++];
        for(int i=1;i<nLen;i+=2)
            res[i]=nums[ans++];
        
        return res;
    }
};
