LeetCode周赛268场

这周的周赛题都比较简单，前两题可以直接暴力和模拟，而第三题也是可以稍微想一下就做出来，简直不要太舒服

2078. 两栋颜色不同且距离最远的房子

这题的暴力解法可以直接写出来

class Solution {
public:
    int maxDistance(vector<int>& colors) {
        int clen=colors.size();
        int res=0;
        for(int i=0;i<clen;++i){
            for(int j=0;j<clen;++j){
                if(colors[i]==colors[j])
                    continue;
                else
                    res=max(res,j-i);
            }
        }
        return res;
    }
};

同时还有一个稍微贪心的做法，分三种情况

  1. 如果最左边和最右边房子颜色不一样，则直接返回(n-1)-0
  2. 固定下标为0位置，向后面找与它最大距离的不同颜色房子
  3. 固定下标为n-1位置，向前找最大距离的不同颜色房子，并与第二步的数值作比较，得出最大距离
class Solution {
    public int maxDistance(int[] colors) {
        int length = colors.length;

        // 如果首位颜色不同直接返回
        if (colors[0] != colors[length - 1]) {
            return length - 1;
        }
        
        // 获取左边第一个不相同的位置
        int left = 1;
        while (colors[left] == colors[0]) {
            left += 1;
        }
        // 获取右边第一个不相同的位置
        int right = length - 2;
        while (colors[right] == colors[length - 1]) {
            right -= 1;
        }

        // 0～right 的长度 和 left～length-1 的长度取最大值
        // 因为要最大，所以不可能在中间，要么就是左边，要么就是右边
        return Math.max(right, length - 1 - left);
    }
}

2079. 给植物浇水

这道题是一个模拟题，直接按照题意做就好了

class Solution {
public:
    int wateringPlants(vector<int>& plants, int capacity) {
        int plen=plants.size();
        if(!plen)
            return 0;
        int fullWater=capacity;//水池
        int res=0;
        for(int i=0;i<plen;++i){
            if(plants[i]<=capacity){
                ++res;
                capacity-=plants[i];

                continue;
            }else{
                //不够水
                res+=(i)+(i+1);
                capacity=fullWater;
                capacity-=plants[i];
                continue;
            }
        }
        return res;
    }
};

2080. 区间内查询数字的频率

这道题需要提前处理一下，存放的容器有两种方案vector和unordered_map

预处理是为了记录每个数在哪些位置出现过

两个版本在query()函数查找都运用了C++算法的内置函数lower_bound()和upper_bound()，这里可以了解一下它们的作用，它们都是运用了二分查找方法，需要用在有特定顺序的容器中

这里举个栗子，在数组vector<int> arr=[10 10 10 20 20 20 30 30]，然后分别调用两个函数，并且第三个参数val填入20，最后输出迭代器的位置iter-arr.begin()

  • lower_bound()：找出容器中第一个大于等于val的数的位置，在这个例子中我们输出迭代器的位置为3
  • upper_bound()：找出容器中第一个大于val的数的位置，在这个例子中输出的迭代器位置为6

unordered_map版本

//unordered_map
class RangeFreqQuery {
private:
    vector<int> arr;
    unordered_map<int,vector<int>> map;
public:
    RangeFreqQuery(vector<int>& vec) {
        this->arr=vec;
        int alen=arr.size();
        for(int i=0;i<alen;++i){
            //预处理
            int cur=arr[i];
            map[cur].push_back(i);//记录每个数字分别出现的位置
        }
    }
    
    int query(int left, int right, int value) {
        vector<int>& curArr=map[value];
        int alen=curArr.size();
        if(!alen)
            return 0;
        
        auto l=lower_bound(curArr.begin(),curArr.end(),left);
        auto r=upper_bound(curArr.begin(),curArr.end(),right);

        return r-l;

    }
};

/**
 * Your RangeFreqQuery object will be instantiated and called as such:
 * RangeFreqQuery* obj = new RangeFreqQuery(arr);
 * int param_1 = obj->query(left,right,value);
 */

vector<vector<int>>版本

class RangeFreqQuery {
  vector<vector<int>> data;

 public:
  RangeFreqQuery(vector<int>& arr) : data(10001) {
    for (int i = 0; i < arr.size(); ++i) {
      data[arr[i]].push_back(i);
    }
  }

  int query(int left, int right, int value) {
    return distance(lower_bound(data[value].begin(), data[value].end(), left),
                    upper_bound(data[value].begin(), data[value].end(), right));
  }
};
