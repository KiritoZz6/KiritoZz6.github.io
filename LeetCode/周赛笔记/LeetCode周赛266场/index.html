class Solution {
public:
    int countVowelSubstrings(string word) {
        int res=0;
        int wlen=word.size();
        for(int i=0;i<wlen;++i){
            char w=word[i];
            if(w=='a'||w=='e'||w=='i'||w=='o'||w=='u'){
                int temp=0;
                bool sig[5]={false,false,false,false,false};
                for(int j=i;j<wlen;++j){
                    w=word[j];
                    if(w=='a'||w=='e'||w=='i'||w=='o'||w=='u'){
                        ++temp;
                        if(w=='a')
                            sig[0]=true;
                        else if(w=='e')
                            sig[1]=true;
                        else if(w=='i')
                            sig[2]=true;
                        else if(w=='o')
                            sig[3]=true;
                        else if(w=='u')
                            sig[4]=true;
                    }
                    else{
                        temp=0;
                        break;}
                    if(temp>=5)
                        if(sig[0]&&sig[1]&&sig[2]&&sig[3]&&sig[4])
                            ++res;
                }
            }
        }

        return res;
    }
};
