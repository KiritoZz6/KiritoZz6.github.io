进程间通信种类

进程间通信方式通常有6种：

  1. 管道
  2. 消息队列
  3. 共享内存
  4. 信号量
  5. 信号
  6. Socket

这些都需要牢牢记住，并且知道他们各自的优缺点以及使用场景

管道

管道分为匿名管道和有名管道

匿名管道代码如下：

//进程间通信之 --- 管道

#include <iostream>
#include <semaphore.h>
#include <unistd.h>
#include <cstring>
#include <sys/wait.h>

using namespace std;


int main(){
    int pipe_arr[2];
    int ret=pipe(pipe_arr);
    if(ret!=0){
        perror("create pipe error!\n");
        exit(-1);
    }

    pid_t pid=fork();
    if(pid==0){
        //子进程
        
        close(pipe_arr[0]);
        const char* message="hello father , this is child!";
        for(int i=0;i<10;++i){
            write(pipe_arr[1],message,strlen(message));
            sleep(1);
        }
        close(pipe_arr[1]);
        exit(0);

    }else if(pid>0){
        //父进程
        printf("fork ok! pid is %d\n",pid);
        close(pipe_arr[1]);
        char buff[64];
        while(true){
            int len=read(pipe_arr[0],buff,sizeof(buff));
            if(len>0){
                printf("this is Father, I read: %s\n",buff);
                memset(buff,'\0',sizeof(buff));
            }else if(len==0){
                printf("father read over!\n");
                
                break;
            }else{
                perror("father read error!\n");
            }

        }
        close(pipe_arr[0]);
        waitpid(pid,NULL,0);
        exit(0);

    }

}

共享内存

共享内存又有两种方式，一个是mmap，一个是shm，这里先放shm的先

shm

//进程间通信之 --- 共享内存
//共享内存有两种方式：mmap、shm


#include <sys/shm.h>
#include <sys/types.h>
#include <fcntl.h>
#include <iostream>
#include <semaphore.h>
#include <unistd.h>
#include <cstring>
#include <sys/wait.h>

#define FILE_NAME "./shm.txt"
#define PROJ_ID 6

using namespace std;

int main(){
    key_t key=ftok(FILE_NAME,PROJ_ID);
    if(key==-1){
        perror("ftok error: ");
    }
    cout<<"key: "<<key<<endl;

    int shm_id=shmget(key,500,IPC_CREAT | 0666);//得到一个共享内存标识符或创建一个共享内存对象并返回共享内存标识符
    if(shm_id==-1){
        perror("shm_id error: ");
        return -1;
    }
    cout<<"shm_id: "<<shm_id<<endl;

    char* buff;
    int pid=fork();

    if(pid==0){
        //子进程
        sleep(2);
        for(int k=0;k<3;k++){
            buff = (char *)shmat(shm_id,NULL,0);  //映射共享内存，得到虚拟地址
            printf("[Message%d]: %s",k+1,buff);
            shmdt(buff);  //解除共享内存
            sleep(1);
        }
    }else if(pid>0){
        //父进程
        for(int i=0;i<3;++i){
            buff=(char*)shmat(shm_id,NULL,0);
            strcpy(buff,"hello son, this is father!!!\n");
            strcat(buff,to_string(i+1).c_str());
            shmdt(buff);//解除共享内存
        }
        
        
        waitpid(pid,NULL,0);
    }
    
    shmctl(shm_id,IPC_RMID,NULL);//删除这片共享内存
    return 0;


}

mmap

//进程间通信之 --- 共享内存（mmap）
//共享内存有两种方式：mmap、shm

//函数原型
//void *mmap(void *addr, size_t length, int prot, int flags, int fd, off_t offset);
/*
* 参数：
*  addr: 指定映射区的首地址。通常传NULL，让内核自动分配
*  length: 映射区的长度，小于等于文件实际长度
*  port: 共享映射区的读写属性。PORT_READ(只读)、PORT_WRITE (只写)、PORT_READ | PORT_WRITE（读写）
*  flags: 共享内存的共享属性。MAP_SHARED、MAP_PRIVATE、MAP_ANONYMOUS (MAP_ANON)
*  fd: 用于创建共享映射区文件的文件描述符
*  offes: 偏移位置，4K的整数倍
*
* 返回值：
* 	成功，返回映射区的首地址
*  	失败：MAP_FILED, 这个宏展开后是：(void *)(-1) ，同时设置errno
*/
//int munmap(void *addr, size_t length);   //释放共享内存映射区


#include <sys/mman.h>
#include <sys/types.h>
#include <fcntl.h>
#include <iostream>
#include <semaphore.h>
#include <unistd.h>
#include <cstring>
#include <sys/wait.h>

#define FILE_NAME "./mmap.txt"
#define BUFFSIZE 512


using namespace std;

int main(){
    int fd=open("mmap.txt",O_RDWR | O_CREAT , 0644);
    if(fd<0){
        perror("open() error: ");
        return -1;
    }
    printf("open ok!\n");

    char* buff=(char*)mmap(NULL,BUFFSIZE,PROT_READ | PROT_WRITE,MAP_SHARED, fd, 0);//建立mmap
    if(buff==NULL){
        perror("mmap error: ");
        return -1;
    }
    close(fd);//关闭fd不影响内存映射

    //测试用例，子进程写，父进程读
    int pid=fork();
    if(pid==0){
        //子进程
        for(int i=0;i<5;++i){
            string message("[Message"+to_string(i+1)+"]: this is son, hello father!\n");
            strcpy(buff,message.c_str());
            printf("NO.%d Message push.\n",i+1);
            sleep(2);
        }
    }else if(pid>0){
        //父进程
        sleep(1);
        for(int i=0;i<5;++i){
            if(buff!=NULL){
                printf("%s",buff);
            }
            sleep(2);
        }
        waitpid(pid,NULL,0);
    }else{
        perror("fork error: ");
        return -1;
    }

    munmap(buff,BUFFSIZE);
    return 0;


}

消息队列

消息队列也有分两种情况，一个是System V的消息队列msg（需要头文件<sys/msg.h>），另个是POSIX的mq（需要头文件<mqueue.h>）

这里只展示msg，需要注意的是msg一定要有个消息结构体（下面代码为struct msg），该结构体还必须有个long类型成员，表示传输类型

//进程间通信之 --- 消息队列
//这里用的是 System V的消息队列，还有POSIX的消息队列需要头文件 <mqueue.h>
/**
 * int msgget(key_t key, int msgflg);//创建或打开msgget()
 * int msgctl(int msqid, int cmd, struct msqid_ds *buf);//控制(删除)msgctl()
 * 
 * *发送消息
 * int msgsnd(int msqid, const void *msgp, size_t msgsz, int msgflg);
 * 
 * *接收
 * ssize_t msgrcv(int msqid, void *msgp, size_t msgsz, long msgtyp,
 */

#include <sys/msg.h>
#include <sys/types.h>
#include <fcntl.h>
#include <iostream>
#include <semaphore.h>
#include <unistd.h>
#include <cstring>
#include <sys/wait.h>

using namespace std;

#define FILE_NAME "./msg.txt"

struct msg{
    long type;
    char message[512];
};

int main(){
    //定义key
    key_t key=ftok(FILE_NAME,3);
    int msgid=msgget(key,IPC_CREAT|0666);
    if(msgid<0){
        perror("msgget error: ");
        return -1;
    }


    struct msg msgs;

    //测试用例，父进程读取消息，子进程发送消息
    int pid=fork();
    if(pid==0){
        //子进程
        for(int i=0;i<10;++i){
            msgs.type=1;
            string message("NO."+to_string(i+1)+" this is son , hello father!");
            strcpy(msgs.message,message.c_str());
            int ret_value = msgsnd(msgid,&msgs,sizeof(struct msg)-sizeof(long),0);//子进程发送消息
            if(ret_value<0){
                perror("son send error: ");
                return -1;
            }
            memset(msgs.message,'\0',sizeof(msgs.message));
            sleep(1);
        }

        //发送over让父进程结束循环
        msgs.type=1;
        strcpy(msgs.message,"over");
        int ret_value=msgsnd(msgid,&msgs,sizeof(struct msg)-sizeof(long),0);
        if(ret_value<0){
            perror("son send over error: ");
            return -1;
        }

    }else if(pid>0){
        //父进程
        int ret_value;
        int count=100;//保底计数器，防止死循环
        while(true){
            
            /*接收消息队列 msgrcv(int msqid, void *msgp, size_t msgsz, long msgtyp, int msgflg);
            * msgtyp：从消息队列内读取的消息形态。如果值为零，
            * 则表示消息队列中的所有消息都会被读取
            * msgflg : 0 表示阻塞等待，IPC_NOWAIT不等待
            */  
            ret_value=msgrcv(msgid,&msgs,sizeof(struct msg)-sizeof(long),0,0);
            if(ret_value<-1){
                perror("father msgrcv error: ");
                exit(-1);
            }else{
                printf("father read[ret:%d type:%ld]: %s\n",ret_value,msgs.type,msgs.message);
            }

            if(strcmp(msgs.message,"over")==0||count==0){
                break;
            }
            --count;
            //sleep(1);
        }
        printf("father break loop!\n");
        waitpid(pid,NULL,0);
    }
    msgctl(msgid,IPC_RMID,0);
    return 0;
}
